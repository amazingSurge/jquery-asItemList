jQuery.asIconList

Examples

You can pass these options as key/value object to $.asIconList() method.

Input

<input type="text" class="default" value='[{"title": "fa-user", "name": "User"}, {"title": "fa-cloud-upload", "name": "Cloud-upload"}, {"title": "fa-search", "name": "Search"}]' name="default"/>
<script>
    $(document).ready(function(){
        $('.default').asItemList({
            sortableID: 'default-list',
            addItem: function() {
                var value = '{"title": "fa-heart", "name": "Heart"}';
                $('<button id="api-addItem-click">addItem</button>').prependTo($("#api-get-click").parent());
                $('#api-addItem-click').val(value);
                
                $('#api-addItem-click').click(function() {
                    var value = $(this).val();
                    $('.default').asItemList('update', value);
                    $('#api-addItem-click').remove();
                });
            }
        });
        
        $('#api-get-click').click(function() {
            var html = '<div>The list is "' + $('.default').asItemList('get') + '"</div>';
            $(html).prependTo($('#api-get-info'));
            return false;
        });
    });
</script>
